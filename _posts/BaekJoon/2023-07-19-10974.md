#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

void Permutation(int arr[], bool visited[], int output[], int depth, int n, int r){
    if(depth == r){
        for(int i = 0; i < depth; i++) cout << output[i];
        cout << "\n"; 
        return;
    }
    else{
        for(int i = 0; i < n; i++){
            if(visited[i] != true){
                visited[i] = true;
                output[depth] = arr[i];
                Permutation(arr, visited, output, depth+1, n, r);
                visited[i] = false;
            }
        }
    }
}

			nums[n] = i + 1;
			Permutation(n + 1);

		}
	}
}
