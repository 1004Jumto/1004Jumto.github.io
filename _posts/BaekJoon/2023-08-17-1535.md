using namespace std;

	ios::sync_with_stdio(false);

	int happy[21]{ 0, };

	for (int i = 1; i <= Testcase; i++) {
		for (int j = 100; j >= strength[i]; j--) {
			dp[j] = max(dp[j], dp[j - strength[i]] + happy[i]);
		}
	}
	for (int i = 0; i < 101; i++) {
		cout << dp[i] << " ";
	}

	cout << dp[100];
	return 0;
