    bool solution(vector<int> arr)
    {
        vector<int> count;
        
        for(int i=0; i < arr.size(); i++) 
            count.push_back(0);
        
        for (int i = 0; i < arr.size(); i++) {
            if (arr[i] <= count.size()) {
                count[arr[i]-1]++;
            }
        }
        
        for (int i = 0; i < arr.size(); i++){
            if(count[i] != 1){
                return false;
            }
        }
        return true;
    }
